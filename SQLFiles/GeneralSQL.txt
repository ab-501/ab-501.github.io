DROP TABLE IF EXISTS Coins;
DROP TABLE IF EXISTS BottleCaps;

CoinsPhpMyAdmin
CREATE TABLE Coins (
  CoinID int AUTO_INCREMENT PRIMARY KEY,
  Country varchar(100) ,
  Currency varchar(1) ,
  Denomination int ,
  Year int ,
  MintMark varchar(10) ,
  Material varchar(20) ,
  Weight int ,
  Diameter int ,
  Thickness int ,
  EdgeType text  ,
  FrontDesign varchar(50) ,
  BackDesign varchar(50) ,
  Designer varchar(50) ,
  Grade varchar(20) ,
  Mintage int ,
  Rarity varchar(20) ,
  History varchar(100) ,
  Notes varchar(100) ,
  Image longblob 
);

CREATE TABLE BottleCaps (
  CapID int AUTO_INCREMENT PRIMARY KEY,
  Brand varchar(50) ,
  BeverageType text ,
  Country varchar(50) ,
  Year int ,
  Material varchar(50) ,
  Colour varchar(50) ,
  Diameter int ,
  Design varchar(100) ,
  Grade text ,
  Rarity text,
  Set text,
  Manufacturer varchar(100) ,
  MarketValue text,
  Notes text,
  Image longblob 
);
